2. 修改 `ExecuteAsync` 方法，添加项目路径处理：

```csharp
public async Task ExecuteAsync(int cardId)
{
    // 获取卡片信息，包含项目关联
    var card = await _executionCardRepositories.FindFirstAsync(
        p => p.Id == cardId,
        p => p.Include(q => q.ProjectRepository)
    );
    
    if (card == null)
    {
        _logger.LogError("Card not found: {CardId}", cardId);
        throw new Exception($"Card not found: {cardId}");
    }
    
    // 确定工作目录
    string workingDirectory = GetWorkingDirectory(card);
    
    // 验证工作目录
    if (!Directory.Exists(workingDirectory))
    {
        _logger.LogError("Working directory does not exist: {WorkingDirectory}", workingDirectory);
        throw new Exception($"Working directory does not exist: {workingDirectory}");
    }
    
    // 构建执行 prompt
    var prompt = BuildExecutionPrompt(card);
    
    // 执行 via Copilot CLI，传递工作目录
    var output = await _copilotCli.ExecutePromptAsync(prompt, workingDirectory);
    
    // 提交结果
    var result = await _submitService.SubmitExecutionResultAsync(
        card.Id,
        (int)ExecutorType.AI,
        "GitHub Copilot CLI",
        output,
        output
    );
    
    _logger.LogInformation(
        "AI execution completed for card {CardId}, Spec passed: {IsPassed}, new status: {NewStatus}",
        card.Id, result.IsSpecPassed, result.NewStatus
    );
}

private string GetWorkingDirectory(ExecutionCard card)
{
    // 如果卡片绑定了项目，使用项目的本地目录
    if (card.ProjectRepositoryId.HasValue && card.ProjectRepository != null && !string.IsNullOrEmpty(card.ProjectRepository.LocalDirectory))
    {
        return card.ProjectRepository.LocalDirectory;
    }
    
    // 否则使用全局配置的工作目录
    return _configuration.GetValue<string>("CopilotCli:WorkingDirectory", "C:\\repo");
}
```
